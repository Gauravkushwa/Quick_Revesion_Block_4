// 1. Total amount spent by each customer
db.orders.aggregate([
  { $group: { _id: "$customerId", totalSpent: { $sum: "$amount" } } }
]);

// 2. Order details with corresponding customer information
db.orders.aggregate([
  {
    $lookup: {
      from: "customers",
      localField: "customerId",
      foreignField: "_id",
      as: "customer"
    }
  },
  { $unwind: "$customer" }
]);

// 3. Orders where the amount is greater than Rs. 500
db.orders.aggregate([
  { $match: { amount: { $gt: 500 } } }
]);

// 4. Average order amount per customer
db.orders.aggregate([
  { $group: { _id: "$customerId", avgAmount: { $avg: "$amount" } } }
]);

// 5. All orders with customer details (ensuring each order has a valid customer)
db.orders.aggregate([
  {
    $lookup: {
      from: "customers",
      localField: "customerId",
      foreignField: "_id",
      as: "customer"
    }
  },
  { $unwind: "$customer" }
]);
